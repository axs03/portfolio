    #include <iostream>
    using namespace std;
    
    //function to print array
    void printArray(int arr[], int size){
        cout << "{";    
        for (int n = 0; n < size; n++) {
            cout << arr[n] << " ";
        }
        cout << "}";
    }
    
    int main() {
        //array declaration
        int numbers[5];
        
        //array declaration 2
        int numbers2[6] = {1,2,4,5,6,7};
    
        cout << "Array 2 : " << printArray(numbers2, 6) << endl;
    
        return 0;
    }
